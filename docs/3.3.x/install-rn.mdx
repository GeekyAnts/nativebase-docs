### Create a new project if not exist

```bash
npx react-native init AwesomeNativeBase
cd AwesomeNativeBase
```

### Install dependencies

<DocTabs
defaultValue='yarn'
values={[
{ label: 'yarn', value: 'yarn' },
{ label: 'npm', value: 'npm' }
]}>

<DocTabItem value="yarn">

```bash
yarn add native-base react-native-svg@12.1.1 react-native-safe-area-context@3.3.2
```

</DocTabItem>

<DocTabItem value="npm">

```bash
npm install native-base react-native-svg@12.1.1 react-native-safe-area-context@3.3.2
```

</DocTabItem>

</DocTabs>

### Run pod install

```bash
cd ios/
pod install
```

### Run the Hello world example

Put the below code in your App.js

```jsx
import React from "react";
import { NativeBaseProvider, Box } from "native-base";

export default function App() {
  return (
    <NativeBaseProvider>
      <Box>Hello world</Box>
    </NativeBaseProvider>
  );
}
```
